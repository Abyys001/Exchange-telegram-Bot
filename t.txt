import requests

def update_price(currency, rate):
    url = "https://pardis.zoxagadget.com/wp-json/pardis/v1/rates"
    data = {
        "currency": currency,
        "rate": rate,
        "api_key": "PX9k7mN2qR8vL4jH6wE3tY1uI5oP0aS9dF7gK2mN8xZ4cV6bQ1wE3rT5yU8iO0pL"
    }
    response = requests.post(url, json=data)
    return response.json()

update_price("GBP", 67000)  # Pound
update_price("USDT", 43000) # Tether
